% Fix Pandoc --citeproc CSL bibliography alignment (esp. [10]+)
\newlength{\cslhangindent}
\setlength{\cslhangindent}{1.5em}

\newlength{\csllabelwidth}
\setlength{\csllabelwidth}{3em} % widen so [10], [100] fit

% CSLReferences environment - add References title, 8pt font per IEEE standard
\newenvironment{CSLReferences}[2]
 {\section*{References}%
  \footnotesize%
  \begin{list}{}{%
    \setlength{\labelwidth}{\csllabelwidth}%
    \setlength{\leftmargin}{\csllabelwidth}%
    \setlength{\itemindent}{-\csllabelwidth}%
    \setlength{\parsep}{0pt}%
    \setlength{\itemsep}{0.5em}%
    \setlength{\rightmargin}{0pt}}}
 {\end{list}}
